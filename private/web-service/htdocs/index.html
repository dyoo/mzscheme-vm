Compiler service

This should not be accessed by hand, but should be accessed programmatically through easyXDM.

Instructions

    var rpc = new easyXDM.Rpc(
	{ remote: "provider.html" },
	{
	    local: {compile : function(version, isModule, name, lang, text) {}},
	    remote: { compile : {}}});

    //
    var compiledProgram = 
        rpc.compile(1, true, 'my program', 'wescheme', '(printf "hello world\n")');
